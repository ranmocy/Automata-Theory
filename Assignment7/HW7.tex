  $$S \rightarrow ABCDS | T_d$$
  $$DA \rightarrow AD$$
  $$DB \rightarrow BD$$
  $$DC \rightarrow CD$$
  $$CA \rightarrow AC$$
  $$CB \rightarrow BC$$
  $$BA \rightarrow AB$$
  $$DT_d \rightarrow T_dd$$
  $$CT_d \rightarrow T_cc$$

  $$BT_c \rightarrow T_bb$$
  $$BT_b \rightarrow T_bb$$
  $$AT_b \rightarrow T_aa$$
  $$AT_a \rightarrow T_aa$$

  $$RL \rightarrow LR$$
  $$BL \rightarrow B$$
  $$RE \rightarrow E$$
  $$YE \rightarrow Ea$$
  $$ZE \rightarrow Ea$$

\end{enumerate}
